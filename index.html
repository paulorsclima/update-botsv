// ============================================================ // SalesIQ — Apps Script (Google Sheets Web App) // Extensões → Apps Script → cole este código → Implantar // ============================================================ function doGet(e) { try { const json = buildJSON(); return ContentService .createTextOutput(JSON.stringify(json)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ erro: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ───────────────────────────────────────────────────────────── function buildJSON() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const wsV = ss.getSheetByName("VENDAS"); const wsE = ss.getSheetByName("ESTOQUE"); const vendas = wsV.getDataRange().getValues(); const estoque = wsE.getDataRange().getValues(); // ── Cabeçalhos ────────────────────────────────────────── const hV = vendas[0]; // linha 1 = header const hE = estoque[0]; const iData = hV.indexOf("Data"); const iSKU = hV.indexOf("SKU"); const iProd = hV.indexOf("Produto"); const iQtd = hV.indexOf("Quantidade"); const iTotal = hV.indexOf("Total"); const iConta = hV.indexOf("Conta"); const iVd = hV.indexOf("Venda/Dia (30d)"); const iV30 = hV.indexOf("Vendas Por 30 Dias"); const iV60 = hV.indexOf("Vendas Por 60 Dias"); const iV90 = hV.indexOf("Vendas Por 90 Dias"); const iESKU = hE.indexOf("SKU"); const iEQtd = hE.indexOf("Quantidade"); // ── Estoque: mapa SKU → quantidade ────────────────────── const estoqueMap = {}; for (let i = 1; i < estoque.length; i++) { const sku = String(estoque[i][iESKU]).trim(); if (sku) estoqueMap[sku] = Number(estoque[i][iEQtd]) || 0; } // ── Processar linhas de VENDAS ─────────────────────────── // Agrupadores const porSKU = {}; // { SKU: { produto, qtd, receita, vd, v30, v60, v90 } } const porMes = {}; // { "YYYY-MM": { receita, pedidos } } const porMesSKU = {}; // { "YYYY-MM": { SKU: { qtd, receita, produto } } } for (let i = 1; i < vendas.length; i++) { const row = vendas[i]; const sku = String(row[iSKU]).trim(); const prod = String(row[iProd]).trim(); const qtd = Number(row[iQtd]) || 0; const total = Number(row[iTotal]) || 0; const vd = Number(row[iVd]) || 0; const v30 = Number(row[iV30]) || 0; const v60 = Number(row[iV60]) || 0; const v90 = Number(row[iV90]) || 0; // Data → "YYYY-MM" let mes = ""; if (row[iData]) { const d = new Date(row[iData]); mes = d.getFullYear() + "-" + String(d.getMonth() + 1).padStart(2, "0"); } if (!sku) continue; // Agrupa por SKU if (!porSKU[sku]) { porSKU[sku] = { produto: prod, qtd: 0, receita: 0, vd: 0, v30: 0, v60: 0, v90: 0, linhas: 0 }; } porSKU[sku].qtd += qtd; porSKU[sku].receita += total; porSKU[sku].vd += vd; porSKU[sku].v30 += v30; porSKU[sku].v60 += v60; porSKU[sku].v90 += v90; porSKU[sku].linhas += 1; // Agrupa por mês if (mes) { if (!porMes[mes]) porMes[mes] = { receita: 0, pedidos: 0 }; porMes[mes].receita += total; porMes[mes].pedidos += 1; if (!porMesSKU[mes]) porMesSKU[mes] = {}; if (!porMesSKU[mes][sku]) porMesSKU[mes][sku] = { qtd: 0, receita: 0, produto: prod }; porMesSKU[mes][sku].qtd += qtd; porMesSKU[mes][sku].receita += total; } } // ── Meses ordenados ───────────────────────────────────── const mesesOrdenados = Object.keys(porMes).sort(); const mesAtual = mesesOrdenados[mesesOrdenados.length - 1] || ""; const mesAnterior = mesesOrdenados[mesesOrdenados.length - 2] || ""; // ── Receita total ──────────────────────────────────────── const receitaTotal = Object.values(porMes).reduce((s, m) => s + m.receita, 0); const pedidosTotal = Object.values(porMes).reduce((s, m) => s + m.pedidos, 0); const ticketMedio = pedidosTotal > 0 ? receitaTotal / pedidosTotal : 0; // Crescimento mês atual vs anterior const recAtual = porMes[mesAtual] ? porMes[mesAtual].receita : 0; const recAnt = porMes[mesAnterior] ? porMes[mesAnterior].receita : 0; const crescimento = recAnt > 0 ? Math.round(((recAtual - recAnt) / recAnt) * 1000) / 10 : 0; // ── Curva ABC ──────────────────────────────────────────── const skus = Object.keys(porSKU); const receitaOrdenada = skus .map(sku => ({ sku, receita: porSKU[sku].receita })) .sort((a, b) => b.receita - a.receita); const receitaGlobal = receitaOrdenada.reduce((s, x) => s + x.receita, 0); let acum = 0; const curvaMap = {}; for (const item of receitaOrdenada) { acum += item.receita; const pct = acum / receitaGlobal; curvaMap[item.sku] = pct <= 0.70 ? "A" : pct <= 0.90 ? "B" : "C"; } // ── Montar lista de produtos ───────────────────────────── const produtos = skus.map(sku => { const d = porSKU[sku]; const est = estoqueMap[sku] !== undefined ? estoqueMap[sku] : null; const vdMed = d.linhas > 0 ? d.vd / d.linhas : 0; const vdFinal = Math.round(vdMed * 100) / 100; const cobertura = est !== null && vdFinal > 0 ? Math.round((est / vdFinal) * 10) / 10 : est === 0 ? 0 : null; const ruptura = est !== null && est <= 0; const status = est === null ? "SEM_EST" : est <= 0 ? "RUPTURA" : cobertura !== null && cobertura < 30 ? "RISCO" : "OK"; const sugestao = est !== null && vdFinal > 0 ? Math.max(0, Math.round(vdFinal * 45 - est)) : 0; const curva = curvaMap[sku] || "C"; // Tendência: v30 do último mês vs 30d anteriores // Usando v60 - v30 como proxy dos 30d anteriores const v30avg = d.v30 / d.linhas; const v60avg = d.v60 / d.linhas; const v30ant = v60avg - v30avg; let trend = 0; if (v30ant > 0) { trend = Math.round(((v30avg - v30ant) / v30ant) * 100); } else if (v30avg > 0) { trend = 100; } return { SKU: sku, produto: d.produto, qtd: d.qtd, receita: Math.round(d.receita * 100) / 100, estoque: est, vd: vdFinal, cobertura: cobertura, ruptura: ruptura, curva: curva, sugestao: sugestao, kpi_trend: trend, v30: Math.round(v30avg * 30), v60: Math.round(v30avg * 60), v90: Math.round(v30avg * 90) }; }); // Ordenar por receita desc produtos.sort((a, b) => b.receita - a.receita); // ── Top 10 ─────────────────────────────────────────────── const top10 = produtos.slice(0, 10).map(p => ({ SKU: p.SKU, produto: p.produto, receita: p.receita, estoque: p.estoque })); // ── Rupturas (estoque = 0 com vendas) ─────────────────── const rupturas = produtos .filter(p => p.ruptura && p.qtd > 0) .map(p => ({ SKU: p.SKU, produto: p.produto, receita: p.receita, estoque: p.estoque })); // ── KPIs Estoque ───────────────────────────────────────── const todosEstoque = Object.keys(estoqueMap); const totalSKUs = todosEstoque.length; const qtdRuptura = todosEstoque.filter(s => estoqueMap[s] <= 0).length; const prodMap = {}; produtos.forEach(p => prodMap[p.SKU] = p); const qtdRisco = todosEstoque.filter(s => { const p = prodMap[s]; return p && p.status === "RISCO"; }).length; const qtdOK = totalSKUs - qtdRuptura - qtdRisco; const qtdCurvaA = produtos.filter(p => p.curva === "A").length; // ── ESTOQUE_DATA (lista completa para aba Estoque) ─────── const estoqueData = todosEstoque.map(sku => { const p = prodMap[sku]; const est = estoqueMap[sku]; const vd = p ? p.vd : 0; const cob = p ? p.cobertura : null; const status = est <= 0 ? "RUPTURA" : (cob !== null && cob < 30) ? "RISCO" : "OK"; const sug = p ? p.sugestao : 0; const curva = p ? p.curva : "C"; const trend = p ? p.kpi_trend : 0; const qtd30 = p ? p.v30 : 0; const rec = p ? p.receita : 0; return { SKU: sku, desc: p ? p.produto : sku, estoque: est, status: status, sugestao: sug, curva: curva, kpi_trend: trend, qtd30: qtd30, receita: rec }; }); // ── vendasPorMes ───────────────────────────────────────── const vendasPorMes = {}; for (const mes of mesesOrdenados) { vendasPorMes[mes] = Object.entries(porMesSKU[mes] || {}).map(([sku, d]) => ({ SKU: sku, produto: d.produto, qtd: d.qtd, receita: Math.round(d.receita * 100) / 100 })); } // ── Insights ───────────────────────────────────────────── // Top alta e queda: mes atual vs anterior const insightsAlta = []; const insightsQueda = []; const estParado = []; const curvaARupt = []; if (mesAtual && mesAnterior) { const skusMesAtual = porMesSKU[mesAtual] || {}; const skusMesAnt = porMesSKU[mesAnterior] || {}; const todosSkus = new Set([...Object.keys(skusMesAtual), ...Object.keys(skusMesAnt)]); const variacoes = []; todosSkus.forEach(sku => { const qAtual = skusMesAtual[sku] ? skusMesAtual[sku].qtd : 0; const qAnt = skusMesAnt[sku] ? skusMesAnt[sku].qtd : 0; if (qAnt > 0 && qAtual > 0) { const pct = Math.round(((qAtual - qAnt) / qAnt) * 100); variacoes.push({ sku, pct, qAtual, curva: curvaMap[sku] || "C", produto: (skusMesAtual[sku] || skusMesAnt[sku]).produto }); } }); variacoes.sort((a, b) => b.pct - a.pct); variacoes.slice(0, 10).forEach(x => insightsAlta.push({ SKU: x.sku, produto: x.produto, qtdAtual: x.qAtual, crescimento: x.pct, curva: x.curva }) ); variacoes.slice(-10).reverse().forEach(x => insightsQueda.push({ SKU: x.sku, produto: x.produto, qtdAtual: x.qAtual, queda: Math.abs(x.pct), curva: x.curva }) ); } produtos.forEach(p => { if (p.cobertura !== null && p.cobertura > 90 && p.estoque > 0) estParado.push({ SKU: p.SKU, produto: p.produto, estoque: p.estoque, cobertura: p.cobertura, curva: p.curva }); if (p.curva === "A" && p.ruptura) curvaARupt.push({ SKU: p.SKU, produto: p.produto, receita: p.receita, sugestao: p.sugestao, curva: p.curva }); }); // ── Montar JSON final ──────────────────────────────────── return { kpis: { receita: Math.round(receitaTotal * 100) / 100, pedidos: pedidosTotal, ticket: Math.round(ticketMedio * 100) / 100, skuRuptura: qtdRuptura, crescimento: crescimento }, monthly: mesesOrdenados.map(m => ({ mes: m, receita: Math.round(porMes[m].receita * 100) / 100 })), produtos: produtos, top10: top10, ruptura: rupturas, vendasPorMes: vendasPorMes, estoque: estoqueData, kpisEstoque: { total: totalSKUs, ruptura: qtdRuptura, risco: qtdRisco, ok: qtdOK, curvaA: qtdCurvaA }, insights: { topAlta: insightsAlta, topQueda: insightsQueda, estoqueParado: estParado, curvaARuptura: curvaARupt } }; }
